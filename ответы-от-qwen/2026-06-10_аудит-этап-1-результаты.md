### 1.6 — Отсутствуют типы в `types.ts` для `WeeksSystem.ts`

**В `WeeksSystem.ts` есть импорт:**
```typescript
import { SpecialWeek, SpecialWeekType, WeeksState } from '../types';
```

**Но в `types.ts` этих типов НЕТ!** → ошибка компиляции.

**Нужно добавить:**
```typescript
/** Тип специальной недели (канон HoMM4) */
export type SpecialWeekType = 
  | 'normal' | 'creature_growth' | 'mana_regen' | 'movement_boost'
  | 'gold_abundance' | 'lucky_week' | 'morale_boost' | 'spell_power'
  | 'defense_week' | 'attack_week' | 'creature_tier_up' | 'necromancy_boost'
  | 'dimension_door' | 'free_hire' | 'experience_boost';

/** Специальная неделя */
export interface SpecialWeek {
  type: SpecialWeekType;
  name: string;
  description: string;
  value: number;
  creatureId?: string;
  faction?: string;
}

/** Состояние системы недель */
export interface WeeksState {
  currentWeek: number;
  currentDay: number;
  specialWeek: SpecialWeek;
  weekHistory: SpecialWeek[];
}
```
